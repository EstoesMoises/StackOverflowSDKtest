## 1. Initialize the SDK

```typescript title="app.ts"
import StackOverflowSDK from 'stackoverflow-sdk';

const sdk = new StackOverflowSDK({
  baseUrl: 'https://api.stackexchange.com',
  accessToken: 'your-access-token'
});
```

## 2. Make Your First API Call

Start exploring Stack Overflow data with these basic examples:

```typescript title="api-calls.ts"
// Fetch recent questions
async function getQuestions() {
  try {
    const questions = await sdk.questions.getQuestions();
    console.log('Recent questions:', questions.items);
    return questions;
  } catch (error) {
    console.error('Failed to fetch questions:', error);
  }
}

// Search for content
async function searchContent(query: string) {
  try {
    const results = await sdk.search.search(query);
    console.log('Search results:', results.items);
    return results;
  } catch (error) {
    console.error('Search failed:', error);
  }
}

// Get user information
async function getUser(userId: number) {
  try {
    const user = await sdk.users.getUserById(userId);
    console.log('User details:', user);
    return user;
  } catch (error) {
    console.error('Failed to fetch user:', error);
  }
}
```

## 5. Complete Example

Here's a complete working example:

```typescript title="complete-example.ts"
import StackOverflowSDK from 'stackoverflow-sdk';

class StackOverflowIntegration {
  private sdk: StackOverflowSDK;

  constructor(accessToken: string) {
    this.sdk = new StackOverflowSDK({
      baseUrl: 'https://api.stackexchange.com',
      accessToken
    });
  }

  async getPopularQuestions() {
    const questions = await this.sdk.questions.getQuestions();
    return questions.items?.slice(0, 10); // Get top 10
  }

  async searchByTag(tag: string) {
    const results = await this.sdk.search.search(`[${tag}]`);
    return results.items;
  }

  async getUserProfile(userId: number) {
    const user = await this.sdk.users.getUserById(userId);
    return user;
  }

  // Team-specific methods
  async getTeamContent(teamId: string) {
    const teamContext = this.sdk.forTeam(teamId);
    
    const [questions, answers] = await Promise.all([
      teamContext.questions.getQuestions(),
      teamContext.answers.getAnswers()
    ]);

    return { questions: questions.items, answers: answers.items };
  }
}

// Usage
const integration = new StackOverflowIntegration('your-access-token');

// Get popular questions
const popularQuestions = await integration.getPopularQuestions();
console.log('Popular questions:', popularQuestions);

// Search for JavaScript content
const jsQuestions = await integration.searchByTag('javascript');
console.log('JavaScript questions:', jsQuestions);

// Get team content
const teamContent = await integration.getTeamContent('team-123');
console.log('Team content:', teamContent);
```

## Error Handling

Always wrap your API calls in try-catch blocks:

```typescript title="error-handling.ts"
async function safeApiCall() {
  try {
    const data = await sdk.questions.getQuestions();
    return data;
  } catch (error) {
    if (error.status === 401) {
      console.error('Authentication failed - check your access token');
    } else if (error.status === 429) {
      console.error('Rate limit exceeded - please retry later');
    } else {
      console.error('API call failed:', error.message);
    }
    throw error;
  }
}
```

## Troubleshooting

**"Invalid access token" errors**
- Verify your access token is valid and not expired
- Ensure you have the necessary permissions for the API endpoint

**API calls return empty results**
- Check that you're using the correct site parameter
- Verify the endpoint supports the parameters you're sending

**Network timeout errors**
- Implement retry logic with exponential backoff
- Consider reducing the page size for large data requests
